(** 

properties of andb {
    andb_true (b:boolean)  : andb Vrai b = b. -straight
    andb_true2 (b:boolean) : andb b Vrai = b. -case 2
    andb_true3 (b:boolean) : andb b b = b. -case 2
}

property of andb { andb_true (b:boolean)  : andb Vrai b = b. -straight }
**)

program ::= 
    | properties of id { properties_descriptor }
    | property of id {assertion_descriptor proofs_helper}
properties_descriptor ::= assertion_descriptor proofs_helper
    | assertion_descriptor proofs_helper; properties_descriptor
assertion_descriptor ::= id args : assertion.
assertion ::= id binOp id
binOp ::= = | <> // straight va pas reagir pareil si c'est un egal ou un different.
args ::= (arg),args | (arg)
arg ::= id : id
proofs_helper ::= straight | case int
i ::= <nat >=2>
id ::= <string>
